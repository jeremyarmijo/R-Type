511 auto& playerComponents = GetRegistry().get_components<PlayerEntity>();
512
513 for (const auto& playerState : playerStates) {

515
516 if (playerId == m_localPlayerId) {
517 if (m_localPlayer < transforms.size() &&
518 transforms[m_localPlayer].has_value()) {

525 playerComponents[m_localPlayer]->current =

528 }
529 } else {
530 auto it = m_otherPlayers.find(playerId);
531 if (it == m_otherPlayers.end()) {
532 SpawnOtherPlayer(playerId, {playerState.posX, playerState.posY});
533 } else {
534 UpdateOtherPlayerPosition(playerId,
535 {playerState.posX, playerState.posY});
536 }
537 }
538 }
539
540 if (m_firstState) {
541 std::unordered_set<uint16_t> activePlayerIds;
542 for (const auto& playerState : playerStates) {
543 activePlayerIds.insert(playerState.playerId);
544 }
545
546 std::vector<uint16_t> toRemove;
547 for (const auto& [playerId, entity] : m_otherPlayers) {
548 if (activePlayerIds.find(playerId) == activePlayerIds.end()) {
549 toRemove.push_back(playerId);
550 }
551 }
552
553 for (uint16_t playerId : toRemove) {
554 RemoveOtherPlayer(playerId);
555 }
556 if (activePlayerIds.find(m_localPlayerId) == activePlayerIds.end() && m_isAlive) {
557 CreateExplosion(m_localPlayer);
558 GetAudio().PlaySound("explosion");
559 m_healthText->SetText("Health: 0");
560 GetRegistry().kill_entity(m_localPlayer);
