230inline TileMap generateSimpleMap(int levelIndex, uint16_t screenWidth = 800,
231 uint16_t screenHeight = 600) {
232 TileMap map;
233 const uint16_t tileSize = 32;
234 const uint16_t mapWidthScreens = 8;
235 uint16_t mapWidth = (screenWidth / tileSize) * mapWidthScreens;
236 uint16_t mapHeight = screenHeight / tileSize;
237
238 map.init(mapWidth, mapHeight, tileSize);
239 map.scrollSpeed = 50.0f + (levelIndex * 10.0f);
240
241 for (int x = 0; x < mapWidth; ++x) {
242 map.setTile(x, 0, TileType::CEILING);
243 map.setTile(x, 1, TileType::CEILING);
244 }
245
246 int groundY = mapHeight - 2;
247 for (int x = 0; x < mapWidth; ++x) {
248 map.setTile(x, groundY, TileType::GROUND);
249 map.setTile(x, groundY + 1, TileType::GROUND);
250 }
251
252 std::cout << "[MapGenerator] Creating map " << mapWidth << "x" << mapHeight
253 << " for level " << levelIndex << std::endl;
254
255 if (levelIndex >= 1) {
256 srand(static_cast<unsigned>(levelIndex * 12345));
257 int numHoles = 3 + levelIndex * 2;
258 for (int i = 0; i < numHoles; ++i) {
259 int holeX = 30 + (rand() % (mapWidth - 60));
260 int holeWidth = 2 + (rand() % 3);
261 for (int dx = 0; dx < holeWidth; ++dx) {
262 map.setTile(holeX + dx, groundY, TileType::EMPTY);
263 map.setTile(holeX + dx, groundY + 1, TileType::EMPTY);
264 }
265 }
266 }
267
268 if (levelIndex >= 1) {
269 int numPlatforms = 2 + levelIndex;
270 for (int i = 0; i < numPlatforms; ++i) {
271 int platX = 40 + (rand() % (mapWidth - 80));
272 int platY = 5 + (rand() % (mapHeight - 10));
273 int platWidth = 3 + (rand() % 4);
274 for (int dx = 0; dx < platWidth; ++dx) {
275 map.setTile(platX + dx, platY, TileType::PLATFORM);
276 }
277 }
278 }
279
280 std::cout << "[MapGenerator] Map generated with " << map.tiles.size()
281 << " tiles (ceiling + ground)" << std::endl;
282
283 return map;
284}
