245inline TileMap generateSimpleMap(int levelIndex, uint16_t screenWidth = 800,
246 uint16_t screenHeight = 600) {
247 TileMap map;
248 const uint16_t tileSize = 32;
249 const uint16_t mapWidthScreens = 8;
250 uint16_t mapWidth = (screenWidth / tileSize) * mapWidthScreens;
251 uint16_t mapHeight = screenHeight / tileSize;
252
253 map.init(mapWidth, mapHeight, tileSize);
254 map.scrollSpeed = 50.0f + (levelIndex * 10.0f);
255
256 for (int x = 0; x < mapWidth; ++x) {
257 map.setTile(x, 0, TileType::CEILING);
258 map.setTile(x, 1, TileType::CEILING);
259 }
260
261 int groundY = mapHeight - 2;
262 for (int x = 0; x < mapWidth; ++x) {
263 map.setTile(x, groundY, TileType::GROUND);
264 map.setTile(x, groundY + 1, TileType::GROUND);
265 }
266
267 std::cout << "[MapGenerator] Creating map " << mapWidth << "x" << mapHeight
268 << " for level " << levelIndex << std::endl;
269
270 if (levelIndex >= 1) {
271 srand(static_cast<unsigned>(levelIndex * 12345));
272 int numHoles = 3 + levelIndex * 2;
273 for (int i = 0; i < numHoles; ++i) {
274 int holeX = 30 + (rand() % (mapWidth - 60));
275 int holeWidth = 2 + (rand() % 3);
276 for (int dx = 0; dx < holeWidth; ++dx) {
277 map.setTile(holeX + dx, groundY, TileType::EMPTY);
278 map.setTile(holeX + dx, groundY + 1, TileType::EMPTY);
279 }
280 }
281 }
282
283 if (levelIndex >= 1) {
284 int numPlatforms = 2 + levelIndex;
285 for (int i = 0; i < numPlatforms; ++i) {
286 int platX = 40 + (rand() % (mapWidth - 80));
287 int platY = 5 + (rand() % (mapHeight - 10));
288 int platWidth = 3 + (rand() % 4);
289 for (int dx = 0; dx < platWidth; ++dx) {
290 map.setTile(platX + dx, platY, TileType::PLATFORM);
291 }
292 }
293 }
294
295 std::cout << "[MapGenerator] Map generated with " << map.tiles.size()
296 << " tiles (ceiling + ground)" << std::endl;
297
298 return map;
299}
